' ===== Oferta, wersje, zamówienie (UC3, UC4, UC7) =====
class Oferta {
  - idOferty: int
  - numerOferty: string
  - status: StatusOfertyEnum
  + generujDokument(): DokumentPDF
  + podpiszOfertę(): bool
  + anulujOferte(): bool

}

enum StatusOfertyEnum {
  DRAFT
  W_NEGOCJACJI
  ZAAKCEPTOWANA
  ANULOWANA
}

}

enum StatusZamowieniaEnum {
  NOWE
  W_REALIZACJI
  ZAMOWIONE
  ZAKONCZONE
  ANULOWANE
}

class Faktura {
  - idFaktury: int
  - numerFaktury: string

  NIEZAPLACONA
  CZESCIOWO_ZAPLACONA
  ZAPLACONA

}
